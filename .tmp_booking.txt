import { test, expect } from '@playwright/test';
import { skipIfMissingSupabaseEnv } from "./test-prerequisites";

skipIfMissingSupabaseEnv();

test.describe('Booking flows', () => {
  test('TC-2.1 Search availability and open CheckoutModal', async ({ page }) => {
    await page.goto('/');
    await page.fill('input[type="date"]', '2026-12-15');
    await page.fill('input[type="time"]', '19:00');
    await page.fill('input[type="number"]', '2');

    await page.click('button:has-text("Search Availability")');
    await expect(
      page.locator('h2:has-text("Availability Results")'),
    ).toBeVisible({ timeout: 15000 });

    const result = page.locator('ul li button');
    if (await result.count() === 0) {
      test.skip(true, 'No availability options were returned for the booking smoke test.');
      return;
    }

    await result.first().click();
    await expect(
      page.locator('h2:has-text("Confirm Your Booking")'),
    ).toBeVisible({ timeout: 5000 });
  });
});
